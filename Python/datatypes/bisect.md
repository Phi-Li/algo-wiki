def bisect_left(A, x, lo=0, hi=None):
    if lo < 0:
        raise ValueError('lo must be non-negative')
    if hi is None:
        hi = len(A)

    while lo < hi:
        mid = (lo + hi) // 2
        # Use __lt__ to match the logic
        #   in list.sort() and in heapq
        if A[mid] < x:
            #  lo              mid
            # [ ... (val < x) ... ] [ ... x
            #                        lo
            lo = mid + 1
        else:
            #  mid               hi
            # [ ... (val >= x) ... ]
            #  hi
            hi = mid

    return lo
```

def bisect_right(A, x, lo=0, hi=None):
    if lo < 0:
        raise ValueError('lo must be non-negative')
    if hi is None:
        hi = len(A)

    while lo < hi:
        mid = (lo + hi) // 2
        # Use __lt__ to match the logic
        #   in list.sort() and in heapq
        if x < A[mid]:
            #            mid              hi
            #   x ... ] [ ... (val > x) ... ]
            #            hi
            hi = mid
        else:
            #  lo               mid
            # [ ... (val <= x) ... ]
            #                        lo
            lo = mid + 1

    return lo
```
